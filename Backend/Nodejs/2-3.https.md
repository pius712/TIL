# http/https

## http/https 비교 

### http

```javascript
const http = require('http');

http.createServer((req,res)=>{
    res.end('http server');
}).listen(80);
```

### https 

```javascript 
const https = require('https');
https.createServer({
    cert: fs.readFileSync('도메인 인증서 경로'),
    key: fs.readFileSync('도메인 비밀키 경로'),
    ca:[
        fs.readFileSync('상위 인증서 경로');
        fs.readFileSync('상위 인증서 경로');
        fs.readFileSync('상위 인증서 경로');
    ]
}, (req,res)=>{
    res.end('https server')
}).listen(443);
```
